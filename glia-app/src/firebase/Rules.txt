//Database Rules
rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    // By default, nobody can read or write anything.
    // This is a secure default.
    match /{document=**} {
      allow read, write: if false;
    }

    // Rule for the 'schedule' collection
    // Anyone (logged in or not) can read the schedule.
    // Only an admin can create, update, or delete events.
    match /schedule/{eventId} {
      allow read: if true;
      allow write: if request.auth.token.email == 'adityamanoja@gmail.com';
    }

    // Rule for the 'notifications' collection
    // Anyone can read notifications.
    // Only an admin can create them.
    match /notifications/{notificationId} {
      allow read: if true;
      allow write: if request.auth.token.email == 'adityamanoja@gmail.com';
    }

    // Rule for each user's personal 'itinerary'
    // Only the authenticated user can read or write to their own itinerary.
    // Users cannot see anyone else's itinerary.
    match /users/{userId}/itinerary/{eventId} {
      allow read, write: if request.auth.uid == userId;
    }
  }
}

//Storage Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /{document=**} {
      // Allow any user who is logged in to read and write all data.
      allow read, write: if request.auth != null;
    }
  }
}
