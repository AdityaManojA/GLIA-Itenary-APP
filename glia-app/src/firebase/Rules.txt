//Database Rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /{document=**} {
      // Allow any user who is logged in to read and write all data.
      allow read, write: if request.auth != null;
    }
  }
}

//Storage Rules

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    match /{allPaths=**} {
      // Allow any user who is logged in to upload and download files.
      allow read, write: if request.auth != null;
    }
  }
}
